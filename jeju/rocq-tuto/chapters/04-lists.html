The List Type

Lists are defined inductively, just like natural numbers:

list.v
(* How list is defined *)
Inductive list (A : Type) : Type :=
  | nil : list A                    (* Empty list *)
  | cons : A -> list A -> list A.   (* Add element to front *)

(* Notation:
   [] for nil
   x :: xs for cons x xs
   [1; 2; 3] for 1 :: 2 :: 3 :: []
*)

Basic List Functions

list_functions.v
Require Import List.
Import ListNotations.

(* Length of a list *)
Fixpoint length {A : Type} (l : list A) : nat :=
  match l with
  | [] => 0
  | _ :: t => S (length t)
  end.

Compute length [1; 2; 3; 4].  (* = 4 *)

(* Append two lists *)
Fixpoint app {A : Type} (l1 l2 : list A) : list A :=
  match l1 with
  | [] => l2
  | h :: t => h :: app t l2
  end.

(* Notation: l1 ++ l2 *)
Compute [1; 2] ++ [3; 4].  (* = [1; 2; 3; 4] *)

(* Reverse a list *)
Fixpoint rev {A : Type} (l : list A) : list A :=
  match l with
  | [] => []
  | h :: t => rev t ++ [h]
  end.

Compute rev [1; 2; 3].  (* = [3; 2; 1] *)

(* Map a function over a list *)
Fixpoint map {A B : Type} (f : A -> B) (l : list A) : list B :=
  match l with
  | [] => []
  | h :: t => f h :: map f t
  end.

Compute map (fun x => x * 2) [1; 2; 3].  (* = [2; 4; 6] *)

Jeju says:

The {A : Type} with curly braces makes A an implicit argument. Rocq figures it out automatically!

Proving Properties of Lists

We use induction on lists, similar to natural numbers:

  • Base case: Prove for []
  • Inductive case: Prove for h :: t, assuming it holds for t
list_proofs.v
Require Import List.
Import ListNotations.

(* Length of append *)
Theorem app_length : forall (A : Type) (l1 l2 : list A),
  length (l1 ++ l2) = length l1 + length l2.
Proof.
  intros A l1 l2.
  induction l1 as [| h t IHt].
  - simpl. reflexivity.
  - simpl. rewrite IHt. reflexivity.
Qed.

(* Append is associative *)
Theorem app_assoc : forall (A : Type) (l1 l2 l3 : list A),
  l1 ++ (l2 ++ l3) = (l1 ++ l2) ++ l3.
Proof.
  intros A l1 l2 l3.
  induction l1 as [| h t IHt].
  - simpl. reflexivity.
  - simpl. rewrite IHt. reflexivity.
Qed.

(* Append with nil *)
Theorem app_nil_r : forall (A : Type) (l : list A),
  l ++ [] = l.
Proof.
  intros A l.
  induction l as [| h t IHt].
  - simpl. reflexivity.
  - simpl. rewrite IHt. reflexivity.
Qed.
